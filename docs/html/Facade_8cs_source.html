60 {
61 Console.WriteLine($"Selecciona 6 Pokémon para {jugador.Nombre}:");
62 List<Pokemon> pokemonsDisponibles = Pokedex.listaPokemons;

64 for (int i = 0; i < pokemonsDisponibles.Count; i++)

67 }
68
69 HashSet<string> seleccionados = new HashSet<string>();
70
71 for (int i = 0; i < 6; i++)
72 {
73 Console.WriteLine($"Selecciona el Pokémon {i + 1} (1-{pokemonsDisponibles.Count}):");
74 int seleccion = int.Parse(Console.ReadLine()) - 1;
75
76 if (seleccion >= 0 && seleccion < pokemonsDisponibles.Count)
77 {
78 string nombrePokemon = pokemonsDisponibles[seleccion].Nombre;

81 {
82 jugador.AgregarPokemon(pokemonsDisponibles[seleccion]);
83 seleccionados.Add(nombrePokemon);
84 }
85 else
86 {
87 Console.WriteLine("Ya has seleccionado este Pokémon. Intenta de nuevo.");
88 i--;
89 }
90 }
91 else
92 {
93 Console.WriteLine("Selección no válida. Intenta de nuevo.");
94 i--;
