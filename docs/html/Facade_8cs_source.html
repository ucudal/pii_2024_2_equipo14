148 public string ComenzarBatalla(string playerDisplayName, string? opponentDisplayName)
149 {
150 // El símbolo ? luego de Trainer indica que la variable opponent puede
151 // referenciar una instancia de Trainer o ser null.
152 Entrenador? opponent;
153
154 if (!OpponentProvided() && !SomebodyIsWaiting())
155 {
156 return "No hay nadie esperando";
157 }
158
159 if (!OpponentProvided() && this.ListaDeEspera.Count > 0)
160 {
161 opponent = this.ListaDeEspera.GetAlguienEsperando(opponentDisplayName);
162
163 // El símbolo ! luego de opponent indica que sabemos que esa
164 // variable no es null. Estamos seguros porque SomebodyIsWaiting
165 // retorna true si y solo si hay usuarios esperando y en tal caso
166 // GetAnyoneWaiting nunca retorna null.
167 return this.CrearBatalla(playerDisplayName, opponent!.Nombre);
168 }
169
170 // El símbolo ! luego de opponentDisplayName indica que sabemos que esa
171 // variable no es null. Estamos seguros porque OpponentProvided hubiera
172 // retorna false antes y no habríamos llegado hasta aquí.
173 opponent = this.ListaDeEspera.EncontrarJugadorPorUsuario(opponentDisplayName!);
174
175 if (!OpponentFound())
176 {
177 return $"{opponentDisplayName} no está esperando";
178 }
179
180 return this.CrearBatalla(playerDisplayName, opponent!.Nombre);
181
182 // Funciones locales a continuación para mejorar la legibilidad
183
184 bool OpponentProvided()
185 {
186 return !string.IsNullOrEmpty(opponentDisplayName);
187 }
188
189 bool SomebodyIsWaiting()
190 {
191 return this.ListaDeEspera.Count != 0;
192 }
193
194 bool OpponentFound()
195 {
196 return opponent != null;
197 }

359 {
360 string mensaje;

366 else
